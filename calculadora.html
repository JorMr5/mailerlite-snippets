<script>

function calcular() {

    const piezasHora = Number(document.getElementById("piezasHora").value);

    const horasDia = Number(document.getElementById("piezasDia").value);

    const scrap = Number(document.getElementById("porcentaje").value) / 100;

    const costeKg = Number(document.getElementById("costeKg").value);


    if (!piezasHora || !horasDia || !scrap || !costeKg) {

        alert("Rellena todos los campos por favor.");

        return;

    }


    // Cálculos

    const scrapHora = piezasHora * scrap;

    const scrapDia = scrapHora * horasDia;

    const scrapMes = scrapDia * 30;

    const scrapAno = scrapMes * 12;


    const costeHora = scrapHora * costeKg;

    const costeDia = scrapDia * costeKg;

    const costeMes = scrapMes * costeKg;

    const costeAno = scrapAno * costeKg;


    // Mostrar resultados

    document.getElementById("resultados").innerHTML = `

        <div class="resultado">

            <h3>Coste por hora</h3>

            <div class="valor">${costeHora.toFixed(2)} €</div>

        </div>


        <div class="resultado">

            <h3>Coste por día</h3>

            <div class="valor">${costeDia.toFixed(2)} €</div>

        </div>


        <div class="resultado">

            <h3>Coste por mes</h3>

            <div class="valor">${costeMes.toFixed(2)} €</div>

        </div>


        <div class="resultado">

            <h3>Coste por año</h3>

            <div class="valor">${costeAno.toFixed(2)} €</div>

        </div>

    `;

}

</script>
